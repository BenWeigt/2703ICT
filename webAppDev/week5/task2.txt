/* a: What are the titles of all blogs with "Japan" in the title? */
SELECT title FROM Blogs WHERE title like '%Japan%';

/* b: What are the titles of all blogs that contain an article with "Japan" in the headline? */
SELECT title
FROM Blogs
INNER JOIN Articles ON Blogs.id = Articles.blog_id AND Articles.heading like '%Japan%';

/* c: What are the titles of all blogs that contain an article written by the blog creator? */
SELECT title
FROM Blogs
INNER JOIN Articles ON Blogs.id = Articles.blog_id AND Articles.author_id = Blogs.creator_id;
